CREATE TABLE users (
	user_id VARCHAR(30) PRIMARY KEY, 
	username VARCHAR(60),
	first_name VARCHAR(30),
	last_name VARCHAR(30),
	email VARCHAR(30),
	join_date DATE
);



CREATE TABLE genres(
	genre_id INT PRIMARY KEY, 
	genre_name VARCHAR(30),
	genre_list VARCHAR(30)
	
);



CREATE TABLE video(
	video_id INT PRIMARY KEY, 
	video_name VARCHAR(30),
	description VARCHAR(30),
	upload_date DATE,
	release_date DATE,
	video__url VARCHAR(60),
	uploader VARCHAR (30)
   
);



CREATE TABLE watched (
	watched_id SERIAL PRIMARY KEY,
	watched_video_list VARCHAR(30),
    video_id_fk INT REFERENCES video(video_id)
                 
);


CREATE TABLE queue (
	queue_id SERIAL PRIMARY KEY,
	queue_name VARCHAR(30),
    queue_video_list VARCHAR(30)
    
                  
                 
);


CREATE TABLE favorite (
	favorite_id SERIAL PRIMARY KEY,
	favorite_video_list VARCHAR(30),
    video_id_fk INT REFERENCES video(video_id)
                  
    
                  
                 
);
